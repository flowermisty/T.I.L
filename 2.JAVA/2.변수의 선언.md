# [JAVA공부] 변수의 선언

```JAVA
public class Ex01 {

 

 public static void main(String[] args) {

  //변수의 선언

   

  boolean a;

  a=true;//대입연산자 a에는 true가 들어있다

  System.out.println(a);

  

  int b;  //int b=-10 --> 같은의미이며, 선언과 동시에 값을 대입하는 결과를 가져온다.

  b=-10;

  System.out.println(b);

   

  float c;

  c=3.1f;

  System.out.println(c);

  

  char d;

  d='a';

  System.out.println(d);

  

  String e;

  e="안녕하세요";

  System.out.println(e);

  

  System.out.println(b+c+d);

   
 }

 

}

  

 

 }

 

}
```
